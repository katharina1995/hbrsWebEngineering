function QueueFifo(x) {
 var content = [];
 var self = {
 addElement: function(element) {
 content.push(element);
 },
 getElement: function() {
 if(self.isEmpty() === false) {
 return content.shift();
 } else {
 throw "Array Empty";
 }
 },
 size: function() {
 return content.length;
 },
 isEmpty: function() {

Schreiben Sie eine JavaScript-Fabrikfunktion für ungeordnete Mengen. In ungeordneten Mengen tritt kein Element mehrfach auf. Privatisieren Sie so viel wie möglich. Schreiben Sie Tests für Ihre Implementierung.

function UnorderedSet(x) {
 var unorderedset = [];
 var self = {
 add: function(element) {
 if(self.has(element) === false) {
 unorderedset.push(element);
 }
 },
 has: function(element) {
 return unorderedset.includes(element);
 },
 delete: function(element) {
 if(self.has(element) === true) {
 unorderedset.splice(unorderedset.indexOf(element),1);
 }
 }
 };
 return self;
 }
